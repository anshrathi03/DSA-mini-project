#include <stdio.h>
#include <string.h>

#define MAX 100

struct Candidate {
    char name[50];
    int votes;
};

int main() {
    struct Candidate candidates[3] = {
        {"ansh", 0},
        {"akhil", 0},
        {"nitin", 0}
    };

    char voters[MAX][50];
    int voterCount = 0;
    int n = 3;

    int choice;
    char voterName[50], voteTo[50];

    do {
        printf("\n1. Vote\n2. Show Results\n3. Exit\n");
        printf("Enter choice: ");
        scanf("%d", &choice);

        switch(choice) {

            case 1: {
                printf("Enter your name: ");
                scanf("%s", voterName);

                int alreadyVoted = 0;

                // Check if voter name matches candidate name (security)
                for(int i = 0; i < n; i++) {
                    if(strcmp(voterName, candidates[i].name) == 0) {
                        printf("Invalid! You cannot use candidate name.\n");
                        alreadyVoted = 1;
                        break;
                    }
                }

                // Check duplicate vote
                for(int i = 0; i < voterCount && !alreadyVoted; i++) {
                    if(strcmp(voters[i], voterName) == 0) {
                        printf("You have already voted!\n");
                        alreadyVoted = 1;
                        break;
                    }
                }

                if(alreadyVoted) break;

                // Show candidate list
                printf("\nAvailable Candidates:\n");
                for(int i = 0; i < n; i++) {
                    printf("- %s\n", candidates[i].name);
                }

                printf("Enter candidate name to vote: ");
                scanf("%s", voteTo);

                int found = 0;

                for(int i = 0; i < n; i++) {
                    if(strcmp(candidates[i].name, voteTo) == 0) {
                        candidates[i].votes++;

                        strcpy(voters[voterCount], voterName);
                        voterCount++;

                        found = 1;
                        printf("Vote cast successfully!\n");
                        break;
                    }
                }

                if(!found) {
                    printf("Invalid candidate!\n");
                }

                break;
            }

            case 2: {
                printf("\n--- Results ---\n");

                int maxVotes = -1;
                char winner[50] = "";

                for(int i = 0; i < n; i++) {
                    printf("%s : %d votes\n", candidates[i].name, candidates[i].votes);

                    if(candidates[i].votes > maxVotes) {
                        maxVotes = candidates[i].votes;
                        strcpy(winner, candidates[i].name);
                    }
                }

                if(maxVotes >= 0)
                    printf("Winner: %s\n", winner);
                else
                    printf("No votes yet!\n");

                break;
            }

            case 3:
                printf("Exiting...\n");
                break;

            default:
                printf("Invalid choice!\n");
        }

    } while(choice != 3);

    return 0;
}
