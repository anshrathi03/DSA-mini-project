#include <stdio.h>
#include <string.h>

#define MAX 100

struct Candidate {
    char name[50];
    int votes;
};

int main() {
    struct Candidate candidates[MAX];
    char voters[MAX][50];

    int n, voterCount = 0;

    printf("Enter number of candidates: ");
    scanf("%d", &n);

    // Input candidates
    for(int i = 0; i < n; i++) {
        printf("Enter candidate name: ");
        scanf("%s", candidates[i].name);
        candidates[i].votes = 0;
    }

    int choice;
    char voterID[50], voteTo[50];

    do {
        printf("\n1. Vote\n2. Show Results\n3. Exit\nEnter choice: ");
        scanf("%d", &choice);

        switch(choice) {
            case 1:
                printf("Enter voter ID: ");
                scanf("%s", voterID);

                // Check duplicate voter
                int alreadyVoted = 0;
                for(int i = 0; i < voterCount; i++) {
                    if(strcmp(voters[i], voterID) == 0) {
                        alreadyVoted = 1;
                        break;
                    }
                }

                if(alreadyVoted) {
                    printf("You have already voted!\n");
                } else {
                    printf("Enter candidate name: ");
                    scanf("%s", voteTo);

                    int found = 0;
                    for(int i = 0; i < n; i++) {
                        if(strcmp(candidates[i].name, voteTo) == 0) {
                            candidates[i].votes++;
                            strcpy(voters[voterCount], voterID);
                            voterCount++;
                            found = 1;
                            printf("Vote cast successfully!\n");
                            break;
                        }
                    }

                    if(!found) {
                        printf("Invalid candidate!\n");
                    }
                }
                break;

            case 2:
                printf("\n--- Results ---\n");

                int maxVotes = 0;
                char winner[50];

                for(int i = 0; i < n; i++) {
                    printf("%s : %d votes\n", candidates[i].name, candidates[i].votes);

                    if(candidates[i].votes > maxVotes) {
                        maxVotes = candidates[i].votes;
                        strcpy(winner, candidates[i].name);
                    }
                }

                printf("Winner: %s\n", winner);
                break;

            case 3:
                printf("Exiting...\n");
                break;

            default:
                printf("Invalid choice!\n");
        }

    } while(choice != 3);

    return 0;
}
